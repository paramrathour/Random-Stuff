Here are formulas of roots of polynomial functions:
\begin{enumerate}
	\item Solution of $ax+b$ is:
	\begin{equation}
		r=-\frac{b}{a}
	\end{equation}
	\item Solution of $ax^2+bx+c$ is:
	\begin{equation}
		r_{1,2}=\frac{-b\pm\sqrt{b^2-4ac}}{2a}\equiv\frac{-2c}{b\pm\sqrt{b^2-4ac}}
	\end{equation}
